CREATE TABLE MERGED GAME DATA  # This SQL code creates a new table called "merged_game_data" by performing a LEFT JOIN between the "gamesales" and "gameconsoles" tables based on the console name. The resulting table includes various columns from both tables, such as game information (name, console, release year, genre, company, sales) and console information (type, company, release year, discontinuation year, units sold, remarks).
 CREATE TABLE merged_game_data AS
 SELECT 
     -- Game info # (you can select specific columns or use gs.* to select all)
     gs.Name AS Game_Name, #    Renaming for clarity
     gs.Console AS Console_Name,
     gs.Year AS Game_Release_Year,
     gs.Genre,
     gs.Company AS Game_Company,
     gs.`North America_Sales`,
     gs.`Europe_Sales`,
     gs.`Japan_Sales`,
     gs.`Global_Sales`,
 
     -- Console info
     gc.Type AS Console_Type,
     gc.Company AS Console_Company,
     gc.`Released Year` AS Console_Release_Year,
     gc.`Discontinuation Year` AS Console_Discontinuation_Year,
     gc.`Units sold (million)` AS Console_Units_Sold,
     gc.Remarks AS Console_Remarks
 
 FROM gamesales gs
 LEFT JOIN gameconsoles gc
     ON gs.Console = gc.`Console Name`
